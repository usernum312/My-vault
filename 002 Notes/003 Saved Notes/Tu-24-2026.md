```js
(async function autoTranslate() {
  const chapterDiv = document.getElementById('LNReader-chapter');
  if (!chapterDiv) return;

  if (chapterDiv.getAttribute('data-translated') === 'true') return;

  const paragraphs = chapterDiv.querySelectorAll('p, div:not([class*="tool"]), h1, h2, h3, h4, h5, h6');
  if (paragraphs.length === 0) return;

  const delay = ms => new Promise(resolve => setTimeout(resolve, ms));

  async function translateText(text, source = 'en', target = 'ar') {
    if (!text || text.trim().length === 0) return text;
    try {
      const url = `https://api.mymemory.translated.net/get?q=${encodeURIComponent(text)}&langpair=${source}|${target}`;
      const response = await fetch(url);
      const data = await response.json();
      if (data.responseData && data.responseData.translatedText) {
        return data.responseData.translatedText;
      } else {
        return text;
      }
    } catch (err) {
      console.warn('Translation error:', err);
      return text;
    }
  }
  for (let el of paragraphs) {
    if (el.querySelector('img, a')) continue;

    const originalText = el.innerText;
    if (!originalText || originalText.trim().length < 3) continue;
    const translated = await translateText(originalText);
    if (translated !== originalText) {
      el.setAttribute('data-original', originalText);
      el.innerText = translated;
    }

    await delay(500);
  }

  chapterDiv.setAttribute('data-translated', 'true');
  console.log('Translation completed.');
})();
```
